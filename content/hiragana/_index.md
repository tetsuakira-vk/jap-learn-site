function speak(text) {
  if (!('speechSynthesis' in window)) return;
  window.speechSynthesis.cancel();
  const u = new SpeechSynthesisUtterance(text);
  u.lang = 'ja-JP';
  u.rate = 0.85;
  window.speechSynthesis.speak(u);
  // Flash the card
  document.querySelectorAll('.kana-card').forEach(function(c) {
    if (c.querySelector('.kc-char') && c.querySelector('.kc-char').textContent === text) {
      c.classList.add('played');
      setTimeout(function(){ c.classList.remove('played'); }, 700);
    }
  });
}
